Q7.Write a function that accepts parameter n and returns factorial of n. 

  function factorial(n) {
    if (n < 0) {
      throw new Error("n must be a non-negative integer");
    }
  
    if (n === 0 || n === 1) {
      return 1;
    } else {
      return n * factorial(n - 1);
    }
  }
